## 📊 Data Structure

### Compliance Report Data
```typescript
interface ComplianceReport {
  report_id: string;
  generated_at: string;
  generated_by: string;
  date_range: {
    from: string;
    to: string;
  };
  overall_score: number;
  tenant_count: number;
  total_validations: number;
  compliance_breakdown: {
    format: number;
    content: number;
    compliance: number;
    quality: number;
  };
  top_violations: Array<{
    rule_id: string;
    rule_name: string;
    violation_count: number;
    affected_tenants: number;
    severity: 'error' | 'warning' | 'info';
  }>;
  tenant_compliance: Array<{
    tenant_id: string;
    tenant_name: string;
    compliance_score: number;
    violations_count: number;
    last_validation: string;
    status: 'compliant' | 'at-risk' | 'non-compliant';
  }>;
}
```

## 🔌 API Endpoints

### Generate Compliance Report
```typescript
POST /api/admin/draft/compliance/reports
Body: {
  from_date: string;
  to_date: string;
  tenant_id?: string; // Optional
  practice_area?: string; // Optional
}
Response: ComplianceReport
```

### Get Compliance Reports List
```typescript
GET /api/admin/draft/compliance/reports
Query: {
  limit?: number;
  offset?: number;
}
Response: {
  reports: ComplianceReport[];
  total: number;
}
```

### Get Specific Report
```typescript
GET /api/admin/draft/compliance/reports/{report_id}
Response: ComplianceReport
```

### Export Report
```typescript
GET /api/admin/draft/compliance/reports/{report_id}/export
Query: {
  format: 'pdf' | 'csv' | 'json';
}
Response: File download
```

## 💻 Implementation Example

### Component Structure
```typescript
'use client'

import { useState, useEffect } from 'react'
import { Card } from '@/components/ui/card'
import { Button } from '@/components/ui/button'
import { DateRangePicker } from '@/components/ui/date-range-picker'

export default function ComplianceReportsPage() {
  const [dateRange, setDateRange] = useState({
    from: new Date(Date.now() - 30 * 24 * 60 * 60 * 1000), // 30 days ago
    to: new Date()
  })
  const [loading, setLoading] = useState(false)
  const [report, setReport] = useState<ComplianceReport | null>(null)
  const [reportsHistory, setReportsHistory] = useState<ComplianceReport[]>([])

  const generateReport = async () => {
    setLoading(true)
    try {
      const response = await fetch('/api/admin/draft/compliance/reports', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          from_date: dateRange.from.toISOString(),
          to_date: dateRange.to.toISOString()
        })
      })
      const data = await response.json()
      setReport(data)
    } catch (error) {
      console.error('Error generating report:', error)
    } finally {
      setLoading(false)
    }
  }

  const loadReportsHistory = async () => {
    try {
      const response = await fetch('/api/admin/draft/compliance/reports')
      const data = await response.json()
      setReportsHistory(data.reports)
    } catch (error) {
      console.error('Error loading reports:', error)
    }
  }

  useEffect(() => {
    loadReportsHistory()
  }, [])

  return (
    <div className="container mx-auto p-6">
      <h1 className="text-3xl font-bold mb-6">Compliance Reports</h1>
      
      {/* Generate Report Section */}
      <Card className="p-6 mb-6">
        <h2 className="text-xl font-semibold mb-4">Generate New Report</h2>
        <DateRangePicker
          from={dateRange.from}
          to={dateRange.to}
          onChange={setDateRange}
        />
        <Button 
          onClick={generateReport} 
          disabled={loading}
          className="mt-4"
        >
          {loading ? 'Generating...' : 'Generate Report'}
        </Button>
      </Card>

      {/* Current Report Display */}
      {report && (
        <Card className="p-6 mb-6">
          <h2 className="text-xl font-semibold mb-4">Current Report</h2>
          {/* Report content */}
        </Card>
      )}

      {/* Reports History */}
      <Card className="p-6">
        <h2 className="text-xl font-semibold mb-4">Report History</h2>
        {/* History list */}
      </Card>
    </div>
  )
}
```
